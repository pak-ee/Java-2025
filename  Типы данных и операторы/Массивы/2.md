Практика №2 — примеры java.util.Arrays

```java
import java.util.Arrays;

public class ArraysDemo {
    public static void main(String[] args) {
        int[] nums = {5, 3, 9, 1};

        // toString(...) — строковое представление массива
        System.out.println(Arrays.toString(nums)); // [5, 3, 9, 1]

        // sort(...) — сортировка по возрастанию
        Arrays.sort(nums);
        System.out.println(Arrays.toString(nums)); // [1, 3, 5, 9]

        // binarySearch(...) — позиция элемента в отсортированном массиве
        int pos = Arrays.binarySearch(nums, 5); // 2
        System.out.println("pos = " + pos);

        // equals(...) — покомпонентное равенство
        boolean same = Arrays.equals(nums, new int[]{1, 3, 5, 9});
        System.out.println("equals = " + same); // true

        // compare(...) — лексикографическое сравнение двух массивов
        int cmp = Arrays.compare(nums, new int[]{1, 3, 5, 9, 10});
        System.out.println("compare = " + cmp); // < 0 (первый короче при одинаковом префиксе)
    }
}
```
